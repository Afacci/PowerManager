110 implicit none
111 
112 !---Declare Local Variables---
113 integer,dimension(nm), intent(in) :: c_
114 integer , intent(in) :: t
115 integer :: i,j,k,l
116 real(kind = prec) :: calore,edisp,eeff,teff,pow
117 
118 !---Function Body
119 
120 thprod = 0
121 if(ntrig.gt.0) then
122  do i=is(it),ie(it)
123  j = c_(i)
124  eeff = etael(j,i)*envcorr(t,i,1)
125  teff = etath(j,i)*envcorr(t,i,2)
126  pow = sp(j,i)*pmax(i)*envcorr(t,i,4)
127  thprod = thprod + teff*pow/eeff
128  enddo
129 endif
130 if(nboi.gt.0) then
131  do i=is(ib),ie(ib)
132  j = c_(i)
133  calore = sp(j,i)*pmax(i)*envcorr(t,i,2)
134  if(pes(i).eq.'heat') then
135  k = esource(i)
136  l = c_(k)
137  pow = sp(l,k)*pmax(k)*envcorr(t,k,4)
138  eeff = etael(l,k)*envcorr(t,k,1)
139  edisp = pow*(1.d0/etael(l,k) - 1.d0)
140  if(calore.gt.edisp) calore = limitrecovery(i,edisp,t)
141  endif
142  thprod = thprod + calore
143  enddo
144 endif
145 return
146 
147 end function thprod

487 real(kind = prec) function limitrecovery(i,valore,t)
488 
489 
490 !--Declare Module usage---
491 use plantvar
492 use inputvar
493 
494 implicit none
495 
496 !---Declare Local Variables---
497 real(kind = prec), intent(in) :: valore
498 integer , intent(in) :: i,t
499 real(kind = prec), parameter :: vsmall = 1.0e-20
500 
501 real(kind = prec),allocatable, dimension(:) :: ein, eout
502 real(kind = prec) :: c, a, b
503 integer :: j,n,iboi
504 
505 !---Function Body
506 
507 iboi = i - is(ib) + 1
508 !n = nEtaB(iBoi)
509 n = nsp(i)
510 allocate(ein(n), eout(n))
511 ein = 0.d0
512 eout = 0.d0
513 
514 do j=1,n
515 ! c = etaB(j,1,iBoi)
516  c = sp(j,i)
517  eout(j)= pmax(i)*c*envcorr(t,i,4)
518 ! eIn(j) = eOut(j)/(etaB(j,2,iBoi)*envCorr(t,i,2))
519  ein(j) = eout(j)/(etath(j,i)*envcorr(t,i,2))
520  if(ein(j).ge.valore) exit
521 enddo
522 
523 !a = envCorr(t,i,2)*(etaB(j,2,iBoi) - etaB(j-1,2,iBoi))/(eOut(j) - eOut(j-1))
524 !b = envCorr(t,i,2)*etaB(j-1,2,iBoi) - a*eOut(j-1)
525 
526 a = envcorr(t,i,2)*(etath(j,i) - etath(j-1,i))/(eout(j) - eout(j-1))
527 b = envcorr(t,i,2)*etath(j-1,i) - a*eout(j-1)
528 limitrecovery = -b*valore/(a*valore - 1.0)
529 
530 deallocate(ein)
531 
532 end function limitrecovery
