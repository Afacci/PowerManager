62 subroutine readtrigen
63 
64 !---Declare Unit usage---
65 use shared
66 use inputvar
67 use filetools
68 use cmdvar
69 use interfaces

71 
72 implicit none
73 
74 !---Declare Local Variables---
75 integer :: genunit = 106
76 character(len=50) :: inputfile = './Input/Trigeneration.inp'
77 logical :: filepresent
78 character(len=100) :: buffer,keyword, value,vector,elements,value_
79 integer :: firstline, i, ninp, line, nrow, n1, n2, x, j, il, nl
80 logical,dimension(17) :: ispresent = .false.
81 integer, dimension(100) :: dummy
82 integer :: error
83 character(len=20), dimension(5) :: param
84 integer :: n, nt, nee, net, nec, np
85 real(kind = prec), allocatable, dimension(:,:) :: matrix
86 
87 !---Check File Presence---
88 inquire(file = inputfile, exist = filepresent)
89 if(.not.filepresent) then
90  call abortexecution(1,2)
91 else
92  open(unit = genunit, file = inputfile)
93 endif
94 
95 !---Skip all the lines befor the keyword "begin".
96 firstline = 0;
97 do
98  read(genunit,100) buffer
99  buffer = adjustl(buffer)
100  firstline = firstline + 1
101  if(buffer(1:5).eq.'begin') exit
102 enddo
103 
104 !---Look for the the "Number" entry that is neede for most of the other entries---
105 buffer = 'Number'
106 call ifindentry(buffer,1,genunit,.true.,dummy(1),ispresent(1))
107 ntrig = dummy(1)
108 if(.not.ispresent(1)) call abortexecution(3,5)
109 call allocatevar(1)
110 
111 !--added because some entries are deactivated.
112 ispresent(3) = .true.
113 ispresent(6) = .true.
114 ispresent(7) = .true.
115 ispresent(11) = .true.
116 
117 !---read the input list---
118 line = firstline
119 do
120  call readkeyword(genunit,.false., keyword,value,error,nl)
121  line = line + nl
122  if(error.eq.1) call abortexecution(0,2,nl)
123  select case(keyword)
124  case('end')
125  exit
126  case('Technology')
127  read(value,*) (tect(i), i=1,ntrig)
128  ispresent(15) = .true.
129  case('Power')
130  read(value,*) (pmaxt(i), i=1,ntrig)
131  ispresent(2) = .true.
132 ! case('DegradationRate')
133 ! read(value,*) (degRateT(i), i=1,nTrig)
134 ! isPresent(3) = .true.
135  case('FuelCost')
136  read(value,*) (fuelcostt(i), i=1,ntrig)
137  ispresent(4) = .true.
138  case('FuelLHV')
139  read(value,*) (fuellhvt(i), i=1,ntrig)
140  ispresent(5) = .true.
141 ! case('Investment')
142 ! read(value,*) (invT(i), i=1,nTrig)
143 ! isPresent(6) = .true.
144 ! case('Lifetime')
145 ! read(value,*) (lifeT(i), i=1,nTrig)
146 ! isPresent(7) = .true.
147  case('OnOffCost')
148  read(value,*) (firecostt(i), i=1,ntrig)
149  ispresent(8) = .true.
150  case('OeMCost')
151  read(value,*) (maintcostt(i), i=1,ntrig)
152  ispresent(9) = .true.
153  case('SetPoint')
154  ispresent(10) = .true.
155  value_ = value
156  do i=1,ntrig
157  nspt(i) = hcount(value_)
158  n2 = index(value_,')') + 1
159  value_ = value_(n2 + 1:)
160  enddo
161  call allocatevar(2)
162  do i = 1, ntrig
163  n1 = index(value,'(') + 1
164  n2 = index(value,')') - 1
165  vector = trim(value(n1:n2))
166  read(vector,*) (spt(j,i), j=1,nspt(i))
167  value = value(n2 + 2:)
168  enddo
169 ! case('Size')
170 ! value_ = value
171 ! isPresent(11) = .true.
172 ! do i=1,nTrig
173 ! nSizeT(i) = hCount(value_)
174 ! n2 = index(value_,')') + 1
175 ! value_ = value_(n2 + 1:)
176 ! enddo
177 ! call allocateVar(3)
178 ! do i = 1, nTrig
179 ! n1 = index(value,'(') + 1
180 ! n2 = index(value,')') - 1

182 ! read(vector,*) (kSizeT(i,j), j=1,nSizeT(i))
183 ! value = value(n2 + 2:)
184 ! enddo
185  case('ElettrEfficiency')

187  backspace(genunit)
188  line = line - 1
189  do i = 1,ntrig
190  netaelt(i) = vcount(genunit,.false.)
191  enddo
192  j = sum(netaelt)
193  call allocatevar(4)
194  call rewunit(genunit,j)
195  do i = 1, ntrig
196  etaelt(:,:,i) = dmatrixread(genunit,netaelt(i),2)
197  enddo
198  line = line + j
199  case('ThermalEfficiency')
200  ispresent(13) = .true.
201  backspace(genunit)
202  line = line - 1
203  do i = 1,ntrig
204  netatht(i) = vcount(genunit,.false.)
205  enddo
206  j = sum(netatht)
207  call allocatevar(5)
208  call rewunit(genunit,j)
209  do i = 1, ntrig
210  etatht(:,:,i) = dmatrixread(genunit,netatht(i),2)
211  enddo
212  line = line + j
213  case('ChillingEfficiency')
214  ispresent(14) = .true.
215  backspace(genunit)
216  line = line - 1
217  do i = 1,ntrig
218  netacht(i) = vcount(genunit,.false.)
219  enddo
220  j = sum(netacht)
221  call allocatevar(6)
222  call rewunit(genunit,j)
223  do i = 1, ntrig
224  etacht(:,:,i) = dmatrixread(genunit,netacht(i),2)
225  enddo
226  line = line + j
227  case('MinUpTime')
228  read(value,*) (minuptimet(i), i=1,ntrig)
229  ispresent(16) = .true.
230  case('MinDownTime')
231  read(value,*) (mindowntimet(i), i=1,ntrig)
232  ispresent(17) = .true.
233  case('Number')
234  continue
235  case(' ')
236  if(verb) call warning(4,2,line=line)
237  case('TempCorrection')
238  read(value,*) (param(i), i=1,5)
239  do i=1,5
240  select case(param(i))
241  case('temp')
242  nt = i
243  case('etaEl')
244  nee = i
245  case('etaTh')
246  net = i
247  case('etaCh')
248  nec = i
249  case('pmax')
250  np = i
251  end select
252  enddo
253  do i=1,ntrig
254  ntct(i) = vcount(genunit,.false.)
255  enddo
256  call allocatevar(23,maxval(ntct))
257  allocate(matrix(maxval(ntct),5))
258  n = sum(ntct)
259  call rewunit(genunit,n)
260  do i=1,ntrig
261  matrix = rnan(rval)
262  matrix = dmatrixread(genunit,ntct(i),5)
263  tempcorrt(:,1,i) = matrix(:,nt)
264  tempcorrt(:,2,i) = matrix(:,nee)
265  tempcorrt(:,3,i) = matrix(:,net)
266  tempcorrt(:,4,i) = matrix(:,nec)
267  tempcorrt(:,5,i) = matrix(:,np)
268  enddo
269  deallocate(matrix)
270  case('PresCorrection')
271  read(value,*) (param(i), i=1,5)
272  do i=1,5
273  select case(param(i))
274  case('pres')
275  nt = i
276  case('etaEl')
277  nee = i
278  case('etaTh')
279  net = i
280  case('etaCh')
281  nec = i
282  case('pmax')
283  np = i
284  end select
285  enddo
286  do i=1,ntrig
287  npct(i) = vcount(genunit,.false.)
288  enddo
289  call allocatevar(24,maxval(npct))
290  allocate(matrix(maxval(npct),5))
291  n = sum(npct)
292  call rewunit(genunit,n)
293  do i=1,ntrig
294  matrix = rnan(rval)
295  matrix = dmatrixread(genunit,npct(i),5)
296  prescorrt(:,1,i) = matrix(:,nt)
297  prescorrt(:,2,i) = matrix(:,nee)
298  prescorrt(:,3,i) = matrix(:,net)
299  prescorrt(:,4,i) = matrix(:,nec)
300  prescorrt(:,5,i) = matrix(:,np)
301  enddo
302  deallocate(matrix)
303  case('AltCorrection')
304  read(value,*) (param(i), i=1,5)
305  do i=1,5
306  select case(param(i))
307  case('alt')
308  nt = i
309  case('etaEl')
310  nee = i
311  case('etaTh')
312  net = i
313  case('etaCh')
314  nec = i
315  case('pmax')
316  np = i
317  end select
318  enddo
319  do i=1,ntrig
320  nact(i) = vcount(genunit,.false.)
321  enddo
322  call allocatevar(25,maxval(nact))
323  allocate(matrix(maxval(nact),5))
324  n = sum(nact)
325  call rewunit(genunit,n)
326  do i=1,ntrig
327  matrix = rnan(rval)
328  matrix = dmatrixread(genunit,nact(i),5)
329  altcorrt(:,1,i) = matrix(:,nt)
330  altcorrt(:,2,i) = matrix(:,nee)
331  altcorrt(:,3,i) = matrix(:,net)
332  altcorrt(:,4,i) = matrix(:,nec)
333  altcorrt(:,5,i) = matrix(:,np)
334  enddo
335  deallocate(matrix)
336  case default
337  if(.not.silent) call warning(1,2,line=line,word=keyword)
338  end select
339 enddo
340 
341 !---check if all the variablea were read---
342 ninp = size(ispresent)
343 do i = 1,ninp
344  if(.not.ispresent(i)) call abortexecution(4,i)
345 enddo
