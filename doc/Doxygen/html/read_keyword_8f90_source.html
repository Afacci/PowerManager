71 subroutine readkeyword (theUnit,rew, keyword, value, error,nRow)
72 
73 !---Declare Module usage---
74 use shared
75 
76 !---Declare local variables---
77 implicit none
78 
79 character(len=*), intent(out) :: keyword, value
80 integer, intent(out), optional :: error
81 integer, intent(out), optional :: nrow
82 integer, intent(in) :: theunit
83 logical, intent(in), optional :: rew
84 character(len=500) :: buffer
85 integer :: ifirst, ilast, nrow_
86 logical :: rew_
87 
88 if(present(rew)) then
89  rew_ = rew
90 else
91  rew_ = .false.
92 endif
93 if(present(error)) error = 0
94 
95 nrow_ = 0
96 do
97  nrow_ = nrow_ + 1
98  read(theunit,100) buffer
99  buffer = adjustl(buffer)
100  if(buffer(1:1).ne.'!')then
101  if(buffer(1:3).eq.'end') then !last useful line --> associate a dummy parater to value
102  keyword = trim(buffer(1:3))
103  value = ' '
104  exit
105  else
106  ifirst = index(buffer,'|')
107  ilast = index(buffer(ifirst+1:),'|') + ifirst
108  if(buffer(1:).eq.' ') then !the line is blank --> read another line
109  continue
110  elseif ((ifirst.eq.ilast).or.ilast.eq.0) then !missing one of the "|"
111  if(present(error)) error = 1 !that should delimit the table
112  keyword = ' ' !dummy values
113  value = ' '
114  exit
115  else
116  keyword = trim(buffer(1:ifirst-1))
117  value = trim(buffer(ifirst+1 : ilast-1))
118  exit
119  endif
120  endif
121  endif
122 enddo
123 !---Rewind the input file
124 if(rew_) call rewunit(theunit,nrow_, .false.)
125 if(present(nrow)) nrow = nrow_
126 
127 return
