64  do i=1,ntrig

67  enddo

74  enddo

79  n = nspt(i)
80  cmax = maxval(spt(:,i))
81  pin = cmax*pmaxt(i)/etaelt_(n,i)
82  ptht = ptht + pin*etatht_(n,i)
83  enddo
84 endif
85 pth = zero
86 if(nboi.gt.0) then
87  do i=1,nboi
88  cmax = maxval(spb(:,i))
89  pth = pth + pmaxb(i)*cmax
90  enddo
91 endif
92 pth = pth + ptht
93 
94 !-evaluation of maximum load for each vector.
95 do i=1,ntime
96  ucht(i) = sum(uch(i,:))
97 enddo
98 ucmax = maxval(ucht)
99 utf = zero
100 uef = zero
101 if(nchi.gt.0) then
102  do i=1,nchi
103  n = nspc(i)
104  if(tecc(i).eq.'Absorption') utf = utf + ucmax/etac_(n,i)
105  if(tecc(i).eq.'Mechanical') uef = uef + ucmax/etac_(n,i)
106  enddo
107 endif
108 do i=1,ntime
109  utht(i) = sum(uth(i,:))
110  uelt(i) = sum(uel(i,:))
111 enddo
112 utmax = maxval(utht) !+ uTf
113 uemax = uef + maxval(uelt)
114 
115 !-abort the execution when one of the loads cannot be satisfied.
116 if(utmax.gt.pth) call abortexecution(i=7,j=1,r1=pth, r2=utmax )
117 if(ucmax.gt.pch) call abortexecution(i=7,j=2,r1=pch, r2=ucmax)
118 if(uemax.gt.pel.and.gridconnection.eq.'StandAlone') call abortexecution(i=7,j=2,r1=pel, r2=uemax)
119 
120 do i=1,ntrig+nboi+nchi

122  if(uptime0(i).gt.0.and.startpoint(i).eq.zero) call abortexecution(15,i,word='upTime',r1=uptime0(i),r2=startpoint(i))
123  if(downtime0(i).eq.0.and.startpoint(i).eq.zero) call abortexecution(15,i,word='DownTime',r1=downtime0(i),r2=startpoint(i))
124  if(downtime0(i).gt.0.and.startpoint(i).gt.zero) call abortexecution(15,i,word='DownTime',r1=downtime0(i),r2=startpoint(i))
125 enddo
