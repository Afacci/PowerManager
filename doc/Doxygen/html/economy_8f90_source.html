96 
97 n = size(uel,2)
98 
99 if(gridconnection.ne.'StandAlone') then
100  p = elprod(c,t)
101  u = 0
102  do i=1,n
103  u = u + uel(t,i)
104  enddo
105  u = u + elselfcons(c,t)
106 endif
107 rvend = zero
108 do i=1,n
109  rvend = rvend + uel(t,i)*cel(t,i)*dt(t)
110 enddo
111 
112 elrev = zero
113 select case(gridconnection)
114  case('StandAlone')
115  elrev = rvend
116  case('Stock')
117  if(p.gt.u) then
118  elrev = rvend + (p - u)*gridsellcost(t)*dt(t)
119  else
120  elrev = rvend + (p - u)*gridbuycost(t)*dt(t)
121  endif
122  case('Sell+Buy')
123  elrev = rvend + (p*gridsellcost(t) - u*gridbuycost(t))*dt(t)
124  case('GSE')
125  enet = (p - u)*dt(t)
126  if(p.gt.u) then
127  rgrid = enet*gridsellcost(t)*k_el
128  rtr = enet*k_tr*c_tr
129  rst = enet*c_st
130  elrev = rvend + rgrid + rtr - rst
131  else
132  elrev = rvend + enet*gridbuycost(t)
133  endif
134 end select
135 
136 return
