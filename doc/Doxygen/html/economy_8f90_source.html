112  case('StandAlone')
113  elrev = rvend
114  case('NetMetering')
115  if(p.gt.u) then
116  elrev = rvend + (p - u)*gridsellcost(t)*dt(t)
117  else
118  elrev = rvend + (p - u)*gridbuycost(t)*dt(t)
119  endif
120  case('DedicatedRetire')
121  elrev = rvend + (p*gridsellcost(t) - u*gridbuycost(t))*dt(t)
122 end select
123
