62 subroutine readtrigen
63 
64 !---Declare Unit usage---
65 use inputvar
66 use filetools
67 use cmdvar
68 use interfaces
69 
70 implicit none
71 
72 !---Declare Local Variables---
73 integer :: genunit = 106
74 character(len=50) :: inputfile = './Input/Trigeneration.inp'
75 logical :: filepresent
76 character(len=100) :: buffer,keyword, value,vector,elements,value_
77 integer :: firstline, i, ninp, line, nrow, n1, n2, x, j, il, nl
78 logical,dimension(17) :: ispresent = .false.
79 integer, dimension(100) :: dummy
80 integer :: error
81 
82 !---Check File Presence---
83 inquire(file = inputfile, exist = filepresent)
84 if(.not.filepresent) then
85  call abortexecution(1,2)
86 else
87  open(unit = genunit, file = inputfile)
88 endif
89 
90 !---Skip all the lines befor the keyword "begin".
91 firstline = 0;
92 do
93  read(genunit,100) buffer
94  buffer = adjustl(buffer)
95  firstline = firstline + 1
96  if(buffer(1:5).eq.'begin') exit
97 enddo
98 
99 !---Look for the the "Number" entry that is neede for most of the other entries---
100 buffer = 'Number'
101 call ifindentry(buffer,1,genunit,.true.,dummy(1),ispresent(1))
102 ntrig = dummy(1)
103 if(.not.ispresent(1)) call abortexecution(3,5)
104 call allocatevar(1)
105 
106 !---read the input list---
107 line = firstline
108 do
109  call readkeyword(genunit,.false., keyword,value,error,nl)
110  line = line + nl
111  if(error.eq.1) call abortexecution(0,2,nl)
112  select case(keyword)
113  case('end')
114  exit
115  case('Technology')
116  read(value,*) (tect(i), i=1,ntrig)
117  ispresent(15) = .true.
118  case('Power')
119  read(value,*) (pmaxt(i), i=1,ntrig)
120  ispresent(2) = .true.
121  case('DegradationRate')
122  read(value,*) (degratet(i), i=1,ntrig)
123  ispresent(3) = .true.
124  case('FuelCost')
125  read(value,*) (fuelcostt(i), i=1,ntrig)
126  ispresent(4) = .true.
127  case('FuelLHV')
128  read(value,*) (fuellhvt(i), i=1,ntrig)
129  ispresent(5) = .true.
130  case('Investment')
131  read(value,*) (invt(i), i=1,ntrig)
132  ispresent(6) = .true.
133  case('Lifetime')
134  read(value,*) (lifet(i), i=1,ntrig)
135  ispresent(7) = .true.
136  case('OnOffCost')
137  read(value,*) (firecostt(i), i=1,ntrig)
138  ispresent(8) = .true.
139  case('OeMCost')
140  read(value,*) (maintcostt(i), i=1,ntrig)
141  ispresent(9) = .true.
142  case('SetPoint')
143  ispresent(10) = .true.
144  value_ = value
145  do i=1,ntrig
146  nspt(i) = hcount(value_)
147  n2 = index(value_,')') + 1
148  value_ = value_(n2 + 2:)
149  enddo
150  call allocatevar(2)
151  do i = 1, ntrig
152  n1 = index(value,'(') + 1
153  n2 = index(value,')') - 1
154  vector = trim(value(n1:n2))
155  read(vector,*) (spt(j,i), j=1,nspt(i))
156  value = value(n2 + 2:)
157  enddo
158  case('Size')
159  value_ = value
160  ispresent(11) = .true.
161  do i=1,ntrig
162  nsizet(i) = hcount(value_)
163  n2 = index(value_,')') - 1
164  value_ = value_(n2 + 2:)
165  enddo
166  call allocatevar(3)
167  do i = 1, ntrig
168  n1 = index(value,'(') + 1
169  n2 = index(value,')') - 1
170  vector = trim(value(n1:n2))
171  read(vector,*) (ksizet(i,j), j=1,nsizet(i))
172  value = value(n2 + 2:)
173  enddo
174  case('ElettrEfficiency')
175  ispresent(12) = .true.
176  backspace(genunit)
177  line = line - 1
178  do i = 1,ntrig
179  netaelt(i) = vcount(genunit,.false.)
180  enddo
181  j = sum(netaelt)
182  call allocatevar(4)
183  call rewunit(genunit,j)
184  do i = 1, ntrig
185  etaelt(:,:,i) = dmatrixread(genunit,netaelt(i),2)
186  enddo
187  line = line + j
188  case('ThermalEfficiency')
189  ispresent(13) = .true.
190  backspace(genunit)
191  line = line - 1
192  do i = 1,ntrig
193  netatht(i) = vcount(genunit,.false.)
194  enddo
195  j = sum(netatht)
196  call allocatevar(5)
197  call rewunit(genunit,j)
198  do i = 1, ntrig
199  etatht(:,:,i) = dmatrixread(genunit,netatht(i),2)
200  enddo
201  line = line + j
202  case('ChillingEfficiency')
203  ispresent(14) = .true.
204  backspace(genunit)
205  line = line - 1
206  do i = 1,ntrig
207  netacht(i) = vcount(genunit,.false.)
208  enddo
209  j = sum(netacht)
210  call allocatevar(6)
211  call rewunit(genunit,j)
212  do i = 1, ntrig
213  etacht(:,:,i) = dmatrixread(genunit,netacht(i),2)
214  enddo
215  line = line + j
216  case('MinUpTime')
217  read(value,*) (minuptimet(i), i=1,ntrig)
218  ispresent(16) = .true.
219  case('MinDownTime')
220  read(value,*) (mindowntimet(i), i=1,ntrig)
221  ispresent(17) = .true.
222  case('Number')
223  continue
224  case(' ')
225  if(verb) call warning(4,2,line=line)
226  case default
227  call warning(1,2,line=line,word=keyword)
228  end select
229 enddo
230 
231 !---check if all the variablea were read---
232 ninp = size(ispresent)
233 do i = 1,ninp
234  if(.not.ispresent(i)) call abortexecution(4,i)
235 enddo
236 
237 close(genunit)
238 100 format(a100)
239 
240 end subroutine readtrigen
