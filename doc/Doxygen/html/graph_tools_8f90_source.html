519  subroutine grapharcs
520 
521  !---Declare Module usage---
522 
523  use inputvar
524  use plantvar
525  use economy
526  use energy
527  use myarithmetic
528  use cmdvar
529 
530  !---Declare Local Variables---
531  implicit none
532 
533  integer :: i,j,t,n1,n2,ni,nf,k, iii, nmax
534  integer, allocatable, dimension(:) :: cnew, cold
535  real(kind=prec) :: deltac, cbatt
536 
537  !predList(:,:) = -1!inan(1)
538  !succList(:,:) = -1!inan(1)
539  !predCost(:,:) = -1000! rnan(rVal)
540  !succCost(:,:) = -1000! rnan(rVal)
541 
542  if(debuggraph) then
543  open(unit=999, file='SuccessorWeightMatrix.dat')
544  open(unit=998, file='SuccessorList.dat')
545  endif
546 
547  nmax = maxval(nt)
548 
549  select case(method)
550  case('Forward')
551  allocate(npre(npoint+1), predcost(npoint+1,nmax))
552  allocate(predlist(0:npoint+1,nmax))
553  !---detect the predecessors for each node---
554  n1 = 1
555  ni = 0
556  do t=1,ntime+1
557  n2 = n1 + nt(t) - 1
558  nf = ni + nt(t-1) - 1
559  do i=n1,n2
560  npre(i) = nt(t-1)
561  predlist(i,1:npre(i)) = (/ (k, k=ni,nf) /)
562  enddo
563  ni = nf + 1
564  n1 = n2 + 1
565  enddo
566  !---associate the cost to each arc in the predecessor list---
567  allocate(cnew(nm), cold(nm))
568  do i=1,npoint+1
569  cnew = pointload(i,:)
570  do j=1,npre(i)
571  k = predlist(i,j)
572  cold = pointload(k,:)
573  predcost(i,j) = pointcost(k) + firecost(cnew,cold)
574  enddo
575  enddo
576 
577  case('Backward', 'Reduced-Backward')
578  allocate(nsuc(0:npoint), succcost(0:npoint,nmax))
579  allocate(succlist(0:npoint+1,nmax))
580  !---detect the successors for each node---
581  n1 = 0
582  ni = 1
583  do t=0,ntime
584  n2 = n1 + nt(t) - 1
585  nf = ni + nt(t+1) - 1
586  do i=n1,n2
587  nsuc(i) = nt(t+1)
588  succlist(i,1:nsuc(i)) = (/ (k, k=ni,nf) /)
589  enddo
590  ni = nf + 1
591  n1 = n2 + 1
592  enddo
593  !---associate the cost to each arc in the successor list---
594  do i=0,npoint
595  cold = pointload(i,:)
596  do j=1,nsuc(i)
597  k = succlist(i,j)
598  cnew = pointload(k,:)
599  select case(obj)
600  case('PEC')
601  deltac = pecpenalty(cnew,cold,pointtime(j))
602  case('Economic')
603  deltac = firecost(cnew,cold)
604  if(elstor) then
605  cbatt = cyclecost(cnew, cold)
606  else
607  cbatt = zero
608  endif
609  end select
610  succcost(i,j) = pointcost(i) + deltac + cbatt
611  enddo
612  if(debuggraph) then
613  write(999,*) 'Node = ', i, 'Successor Weight = ' , succcost(i,:)
614  write(998,*) 'Node = ', i, 'Suuccessor List = ' , succlist(i,:)
615  endif
616  enddo
617  end select
618 
619  if(debuggraph) then
620  close(999)
621  close(998)
622  endif
623 
624  deallocate(cnew,cold)
625 
626  if(.not.silent) then
627  print*
628  print*, ' ------------Graph report------------'
629  print*, ' | Plant State number: ', ncomb, '|'
630  print*, ' | Time-steps : ', ntime, '|'
631  print*, ' | Verteces number : ', npoint, '|'
632  if(method.eq.'Forward') print*, ' | Arcs number : ', sum(npre(:)), '|'
633  if(method.eq.'Backward'.or.method.eq.'Reduced-Backward') print*, ' | Arcs number : ', sum(nsuc(:)), '|'
634  print*, ' ------------------------------------'
635  print*
636  endif
637 
638  return
639 
640  end subroutine grapharcs
